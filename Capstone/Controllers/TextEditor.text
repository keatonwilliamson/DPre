    Oscillator2WaveformDegrees INTEGER NOT NULL,
    Oscillator3WaveformValue VARCHAR(55) NOT NULL,
    Oscillator3WaveformDegrees INTEGER NOT NULL,

    Oscillator1VolumeDegrees INTEGER NOT NULL,
    Oscillator2VolumeValue INTEGER NOT NULL,
    Oscillator2VolumeDegrees INTEGER NOT NULL,
    Oscillator3VolumeValue INTEGER NOT NULL,
    Oscillator3VolumeDegrees INTEGER NOT NULL,
    Oscillator1 BIT NOT NULL DEFAULT(0),
